# NOTA: Per usare i collegamenti seriali tra i router (WAN), i Router 2620XM devono essere spenti, bisogna aggiungere il modulo "WIC-1T" o "WIC-2T" negli slot liberi, e poi riaccenderli.

# Collega i Router tra loro con cavo Seriale (Serial DCE)
Connections > Serial DCE
# Collega R1 (es. Serial0/0) a R2 (es. Serial0/0)
# Collega R2 (es. Serial0/1) a R3 (es. Serial0/0)
# Collega R3 (es. Serial0/1) a R1 (es. Serial0/1) - OPZIONALE per chiudere l'anello

# Configurazione interfacce Seriali su R1
# Crea PC temporaneo e collega RS232 del PC a Console del Router R1
Desktop > Terminal > ok

# Router R1
R1>			enable
R1#			configure terminal
R1(config)#		interface Serial0/0
R1(config-if)#		ip address 10.0.0.1 255.255.255.0
# (Solo se questo lato è DCE, cosa che vedi con il comando 'show controllers s0/0' da modalità privilegiata):
R1(config-if)#		clock rate 64000
R1(config-if)#		no shutdown
R1(config-if)#		exit

R1(config)#		interface Serial0/1
R1(config-if)#		ip address 10.0.1.1 255.255.255.0
# (Imposta il clock rate se questo lato è DCE)
R1(config-if)#		no shutdown
R1(config-if)#		exit

# Abilitazione del protocollo di routing RIP su R1
R1(config)#		router rip
R1(config-router)#	network 10.0.0.0
R1(config-router)#	network 10.0.1.0
R1(config-router)#	network 192.168.0.0
R1(config-router)#	network 192.168.1.0
R1(config-router)#	exit
R1(config)#		exit

# Verifica della tabella di routing
R1#			show ip route
